20      private static final String SPACES_OR_PUNCTUATION = "\\p{Punct}|\\p{Blank}";
21  
22      @Override
23      public <T> Query build(FullTextEntityManager fullTextEntityManager, TermSelector<T> termSelector, Class<? extends T> type) {
24          QueryBuilder builder = fullTextEntityManager.getSearchFactory().buildQueryBuilder().forEntity(type).get();
25  
26          BooleanJunction<?> context = builder.bool();
27          boolean valid = false;
28          if (termSelector.isNotEmpty()) {
29              boolean hasTerms = false;
30              BooleanJunction<?> termContext = builder.bool();
31              for (String selected : termSelector.getSelected()) {
32                  if (isNotBlank(selected)) {
33                      BooleanJunction<?> splitContext = builder.bool();
34                      for (String value : selected.split(SPACES_OR_PUNCTUATION)) {
35                          if (isNotBlank(value)) {
36                              BooleanJunction<?> valueContext = builder.bool();
37                              if (termSelector.getSearchSimilarity() != null) {
38                                  valueContext.should(builder.keyword().fuzzy() //
39                                          .withEditDistanceUpTo(termSelector.getSearchSimilarity()) //
40                                          .onField(termSelector.getPath()) //
41                                          .matching(value).createQuery());
42                              }
43                              valueContext.should(builder.keyword() //
44                                      .onField(termSelector.getPath()) //
45                                      .matching(value).createQuery());
46                              valueContext.should(builder.keyword().wildcard() //
47                                      .onField(termSelector.getPath()) //
48                                      .matching("*" + value + "*").createQuery());
49                              if (termSelector.isOrMode()) {
50                                  splitContext.should(valueContext.createQuery());
51                              } else {
52                                  splitContext.must(valueContext.createQuery());
53                              }
54                              hasTerms = true;
55                          }
56                      }
57                      if (hasTerms) {
58                          if (termSelector.isOrMode()) {
59                              termContext.should(splitContext.createQuery());
60                          } else {
61                              termContext.must(splitContext.createQuery());
62                          }
63                      }
64                  }
65              }
66              if (hasTerms) {
67                  context.must(termContext.createQuery());
68                  valid = true;
69              }
70          }
71          try {
72              if (valid) {
73                  return context.createQuery();
74              } else {
75                  return builder.all().except(builder.all().createQuery()).createQuery();
76              }
77          } catch (Exception e) {
78              throw propagate(e);
79          }
80      }
